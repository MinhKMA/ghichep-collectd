apt-get install git
git clone https://github.com/Ennexa/seyren-ubuntu-init-script.git

cd seyren-ubuntu-init-script

cp init.d/seyren /etc/init.d/seyren
chmod +x /etc/init.d/seyren
cp defaults/seyren /etc/default/
